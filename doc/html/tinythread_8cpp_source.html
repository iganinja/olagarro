40 // NOTE 1: The Win32 implementation of the condition_variable class is based on
41 // the corresponding implementation in GLFW, which in turn is based on a
42 // description by Douglas C. Schmidt and Irfan Pyarali:
43 // http://www.cs.wustl.edu/~schmidt/win32-cv-1.html
44 //
45 // NOTE 2: Windows Vista actually has native support for condition variables
46 // (InitializeConditionVariable, WakeConditionVariable, etc), but we want to
47 // be portable with pre-Vista Windows versions, so TinyThread++ does not use
48 // Vista condition variables.

57 condition_variable::condition_variable() : mWaitersCount(0)
58 {

66 condition_variable::~condition_variable()

74 #if defined(_TTHREAD_WIN32_)
75 void condition_variable::_wait()
76 {
77  // Wait for either event to become signaled due to notify_one() or
78  // notify_all() being called
79  int result = WaitForMultipleObjects(2, mEvents, FALSE, INFINITE);
80 
81  // Check if we are the last waiter
82  EnterCriticalSection(&mWaitersCountLock);
83  -- mWaitersCount;
84  bool lastWaiter = (result == (WAIT_OBJECT_0 + _CONDITION_EVENT_ALL)) &&
85  (mWaitersCount == 0);
86  LeaveCriticalSection(&mWaitersCountLock);
87 
88  // If we are the last waiter to be notified to stop waiting, reset the event
89  if(lastWaiter)
90  ResetEvent(mEvents[_CONDITION_EVENT_ALL]);
91 }
92 #endif

94 #if defined(_TTHREAD_WIN32_)
95 void condition_variable::notify_one()
96 {
97  // Are there any waiters?
98  EnterCriticalSection(&mWaitersCountLock);
99  bool haveWaiters = (mWaitersCount > 0);
100  LeaveCriticalSection(&mWaitersCountLock);
101 
102  // If we have any waiting threads, send them a signal
103  if(haveWaiters)
104  SetEvent(mEvents[_CONDITION_EVENT_ONE]);
105 }
106 #endif

123 //------------------------------------------------------------------------------
124 // POSIX pthread_t to unique thread::id mapping logic.
125 // Note: Here we use a global thread safe std::map to convert instances of
126 // pthread_t to small thread identifier numbers (unique within one process).
127 // This method should be portable across different POSIX implementations.
128 //------------------------------------------------------------------------------
129 
130 #if defined(_TTHREAD_POSIX_)
131 static thread::id _pthread_t_to_ID(const pthread_t &aHandle)
132 {
133  static mutex idMapLock;
134  static std::map<pthread_t, unsigned long int> idMap;
135  static unsigned long int idCount(1);
136 
137  lock_guard<mutex> guard(idMapLock);
138  if(idMap.find(aHandle) == idMap.end())
139  idMap[aHandle] = idCount ++;
140  return thread::id(idMap[aHandle]);
141 }
142 #endif // _TTHREAD_POSIX_

156 // Thread wrapper function.
157 #if defined(_TTHREAD_WIN32_)
158 unsigned WINAPI thread::wrapper_function(void * aArg)
159 #elif defined(_TTHREAD_POSIX_)
160 void * thread::wrapper_function(void * aArg)
161 #endif
162 {
163  // Get thread startup information
164  _thread_start_info * ti = (_thread_start_info *) aArg;
165 
166  try
167  {
168  // Call the actual client thread function
169  ti->mFunction(ti->mArg);
170  }
171  catch(...)
172  {
173  // Uncaught exceptions will terminate the application (default behavior
174  // according to C++11)
175  std::terminate();
176  }
177 
178  // The thread is no longer executing
179  lock_guard<mutex> guard(ti->mThread->mDataMutex);
180  ti->mThread->mNotAThread = true;
181 
182  // The thread is responsible for freeing the startup information
183  delete ti;
184 
185  return 0;
186 }

188 thread::thread(void (*aFunction)(void *), void * aArg)
189 {
190  // Serialize access to this thread structure
191  lock_guard<mutex> guard(mDataMutex);
192 
193  // Fill out the thread startup information (passed to the thread wrapper,
194  // which will eventually free it)
195  _thread_start_info * ti = new _thread_start_info;
196  ti->mFunction = aFunction;
197  ti->mArg = aArg;
198  ti->mThread = this;
199 
200  // The thread is now alive
201  mNotAThread = false;
202 
203  // Create the thread
204 #if defined(_TTHREAD_WIN32_)
205  mHandle = (HANDLE) _beginthreadex(0, 0, wrapper_function, (void *) ti, 0, &mWin32ThreadID);
206 #elif defined(_TTHREAD_POSIX_)
207  if(pthread_create(&mHandle, NULL, wrapper_function, (void *) ti) != 0)
208  mHandle = 0;
209 #endif
210 
211  // Did we fail to create the thread?
212  if(!mHandle)
213  {
214  mNotAThread = true;
215  delete ti;
216  }
217 }

247 {

272 unsigned thread::hardware_concurrency()
273 {
274 #if defined(_TTHREAD_WIN32_)
275  SYSTEM_INFO si;
276  GetSystemInfo(&si);
277  return (int) si.dwNumberOfProcessors;
278 #elif defined(_SC_NPROCESSORS_ONLN)
279  return (int) sysconf(_SC_NPROCESSORS_ONLN);
280 #elif defined(_SC_NPROC_ONLN)
281  return (int) sysconf(_SC_NPROC_ONLN);
282 #else
283  // The standard requires this function to return zero if the number of
284  // hardware cores could not be determined.
285  return 0;
