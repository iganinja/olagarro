43 template<typename HostClass>
44 class Thread
45 {
46 public:
47  Thread(const std::string& name) :
48  mName(name),
49  mRunning(false)
50  {
51  }
52 
53  virtual ~Thread()
54  {
55  join();
56  }
57 
58  void start()
59  {
60 #ifdef OLAGARRO_LOG
61  std::cout << "[" << name() << "]" << "Thread: Creating thread" << std::endl;
62 #endif
63  mTThread.reset(new tthread::thread(threadMethod, this));
64 
65  // Wait until threadMethod is running in its own thread
66  while(!isRunning())
67  {
68 #ifdef OLAGARRO_LOG
69  std::cout << "[" << name() << "]" << "Thread: Waiting for thread creation" << std::endl;
70 #endif
71  tthread::this_thread::sleep_for(tthread::chrono::milliseconds(10));
72  }
73 
74 #ifdef OLAGARRO_LOG
75  std::cout << "[" << name() << "]" << "Thread: Thread created" << std::endl;
76 #endif
77  }
78 
79  bool isRunning()
80  {
81  return mRunning;
82  }
83 
84  void join()
85  {
86  if(!mTThread.isNull())
87  {
88  mTThread->join();
89  }
90  }
91 
92  const std::string& name() const
93  {
94  return mName;
95  }
96 
97 protected:
98  virtual void run() = 0;
99 
100 private:
101  static void threadMethod(void* param)
102  {
103  Thread<HostClass>* instance = reinterpret_cast<Thread<HostClass>*>(param);
104 
105  instance->mRunning = true;
106 
107  instance->run();
108 
109  instance->mRunning = false;
110  }
111 
112  Shared<tthread::thread> mTThread;
113  std::string mName;
114  bool mRunning;
115 };
