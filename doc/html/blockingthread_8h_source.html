41 template<typename HostClass>
42 class BlockingThread : public Thread<HostClass>
43 {
44 public:
45  BlockingThread(const std::string& name) :
46  Thread<HostClass>(name),
47  mFinishThread(false),
48  mWorking(false)
49  {
50  }
51 
52  ~BlockingThread()
53  {
54  finish();
55  }
56 
60  void launch()
61  {
62  Thread<HostClass>::start();
63  }
64 
65  void run()
66  {
67  preJobTasks();
68 
69  while(!mFinishThread)
70  {
71  tthread::lock_guard<tthread::mutex> guard(mJobMutex);
72  mJobStartCondVar.wait(mJobMutex);
73 
74  if(mFinishThread)
75  {
76  break; // If finish() has been called while we were blocked
77  }
78 
79  performJob();
80 
81  performBeforeBlocking();
82 
83  mWorking = false;
84  }
85 
86  postJobTasks();
87  }
88 
89  void resumeJob()
90  {
91  assert(Thread<HostClass>::isRunning() && "BlockingThread::resumeJob(): thread is not running");
92 
93  mWorking = true;
94 
95  mJobStartCondVar.notify_all();
96  }
97 
101  void waitForJob()
102  {
103  if(!Thread<HostClass>::isRunning())
104  {
105  return; // Thread already finished, just return
106  }
107 
108  // Just try to lock mJobMutex. If thread is working then we get blocked until it finishes, otherwise thread is waiting and we can exit that method
109  tthread::lock_guard<tthread::mutex> guard(mJobMutex);
110  }
111 
115  void finish()
116  {
117  if(!Thread<HostClass>::isRunning())
118  {
119  return;
120  }
121 
122  mFinishThread = true;
123 
124  mJobStartCondVar.notify_all();
125 
126  Thread<HostClass>::join();
127  }
128 
129  bool isWorking() const
130  {
131  return mWorking;
132  }
133 
134 protected:
138  virtual void preJobTasks() {}
142  virtual void performJob() = 0;
147  virtual void performBeforeBlocking() {}
151  virtual void postJobTasks() {}
152 
153 private:
154  tthread::mutex mJobMutex;
155  tthread::mutex mWaitMutex;
156  mutable tthread::mutex mWorkingMutex;
157  tthread::condition_variable mJobStartCondVar;
158  tthread::condition_variable mWaitCondVar;
159  bool mFinishThread;
160  bool mWorking;
161 };
